#include <stdio.h>

void abs_array(int *array, int n) {
    int count = 0; // инициализируем счетчик негативных чисел
    for (int i = 0; i < n; i++) { // проходим по всем элементам массива
        if (*(array+i) < 0) { // если элемент отрицательный 
            *(array+i) = -*(array+i); // берем его по модулю 
            count++; // и увеличиваем счетчик негативных чисел
        }
    }
    printf("%d ", count); // выводим количество отрицательных чисел в массиве
}

int main() {
    int n;
    printf("введите размер массива:");
    scanf("%d", &n); //вводим рамзер массива
    int array[n]; // создаем массив нужного размера 
    for (int i = 0; i < n; i++) {
        scanf("%d", &array[i]); // вводим n чисел и сохраняем их в массиве 
    }
    abs_array(array, n); //вызываем функцию, которая преобразует отрицательные числа в массиве и подсчитывает их количество
    for (int i = 0; i < n; i++) {
        printf("%d ", array[i]); //выводим массив с преобразованными числами
    }
}
